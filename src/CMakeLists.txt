find_package (PkgConfig)

# Add all your dependencies to the list below
pkg_check_modules (DEPS REQUIRED
    gtk+-3.0>=3.12
    switchboard-2.0
    granite
)

add_definitions (${DEPS_CFLAGS})
link_libraries (${DEPS_LIBRARIES})
link_directories (${DEPS_LIBRARY_DIRS})

find_package (Vala REQUIRED)
include (ValaVersion)
ensure_vala_version ("0.22.0" MINIMUM)

include (ValaPrecompile)
# Add all your vala files and requires packages to the List below to include them in the build
vala_precompile (VALA_C ${PLUGNAME}
    Plug.vala
    Widgets/AppEntry.vala
    Widgets/AppList.vala
    Widgets/AppSettingsView.vala
    Widgets/Footer.vala
    Widgets/MainView.vala
    Widgets/SettingsHeader.vala
    Widgets/SettingsOption.vala
    Widgets/Sidebar.vala
    Backend/NotifySettings.vala
    Backend/NotifyManager.vala
    Backend/App.vala
    ${CMAKE_CURRENT_BINARY_DIR}/config.vala
PACKAGES
    switchboard-2.0
    granite
OPTIONS
    --target-glib=2.32
)

include (GResource)
glib_compile_resources (GLIB_RESOURCES_IMAGES SOURCE ../data/switchboard-plug-notifications.gresource.xml)

add_library (${PLUGNAME} MODULE ${VALA_C} ${GLIB_RESOURCES_IMAGES})
target_link_libraries (${PLUGNAME} ${DEPS_LIBRARIES})

# Installation
install (TARGETS ${PLUGNAME} DESTINATION ${PKGDATADIR})
